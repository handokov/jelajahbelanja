     9→/**
    10→ * Extract shopid and itemid from various Shopee URL formats:
    11→ * - https://shopee.co.id/product-name-i.SHOPID.ITEMID
    12→ * - https://shopee.co.id/product-name-i.SHOPID.ITEMID?...
    13→ * - https://shope.ee/xxx (short URL, will be resolved)
    14→ */
    15→function extractIds(url: string): { shopId: number; itemId: number } | null {
    16→  // Pattern: -i.SHOPID.ITEMID at end of path
    17→  const match = url.match(/-i\.(\d+)\.(\d+)/);
    18→  if (match) {
    19→    return { shopId: parseInt(match[1], 10), itemId: parseInt(match[2], 10) };
    20→  }
    21→  return null;
    22→}

    35→      signal: AbortSignal.timeout(10000),
    36→    });

    38→  } catch {

    43→/**
    44→ * Fetch product data using Shopee's internal API v4
    45→ */
    46→async function fetchShopeeApi(shopId: number, itemId: number): Promise<Record<string, unknown> | null> {
    47→  const apiUrl = `https://shopee.co.id/api/v4/item/get?itemid=${itemId}&shopid=${shopId}`;
    48→
    49→  try {
    50→    const res = await fetch(apiUrl, {
    51→      headers: {
    52→        "User-Agent":
    53→          "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/125.0.0.0 Safari/537.36",
    54→        Accept: "application/json",
    55→        "Accept-Language": "id-ID,id;q=0.9,en-US;q=0.8,en;q=0.7",
    56→        "X-Shopee-Language": "id",
    57→        "X-API-SOURCE": "pc",
    58→        Referer: `https://shopee.co.id/`,
    59→        "af-ac-enc-dat": "null",
    60→      },
    61→      signal: AbortSignal.timeout(15000),
    62→    });
    63→
    64→    if (!res.ok) {
    65→      console.error("[scrape-shopee] Shopee API status:", res.status);
    66→      return null;
    67→    }
    68→
    69→    const data = await res.json();
    70→
    71→    if (data.error) {
    72→      console.error("[scrape-shopee] Shopee API error:", data.error, data.msg || "");
    73→      return null;
    74→    }
    75→
    76→    const item = data.data;
    77→    if (!item) return null;
    78→
    79→    const product: Record<string, unknown> = {};
    80→
    81→    // Title
    82→    product.title = item.name || "";
    83→
    84→    // Image - Shopee returns image field as "xxx" which needs prefix
    85→    if (item.image) {
    86→      product.image = `https://down-id.img.susercontent.com/file/${item.image}`;
    87→    } else if (item.images && Array.isArray(item.images) && item.images.length > 0) {
    88→      product.image = `https://down-id.img.susercontent.com/file/${item.images[0]}`;
    89→    }
    90→
    91→    // Price - Shopee API returns prices in minor units (need to handle)
    92→    // price_min and price_max are in the smallest currency unit
    93→    // For IDR, Shopee sometimes returns the actual price (no conversion needed for IDR)
    94→    const priceMin = item.price_min || item.price;
    95→    const priceMax = item.price_max || item.price;
    96→
    97→    if (priceMin) {
    98→      // Shopee API returns price as array [min, max] or single number
    99→      const priceVal = Array.isArray(priceMin) ? priceMin[0] : priceMin;
   100→      // If price is unreasonably large (above 10M IDR per unit), it might be in minor units
   101→      product.price = typeof priceVal === "number" ? priceVal : parseInt(String(priceVal), 10);
   102→    }
   103→
   104→    // Original price (before discount)
   105→    const priceBeforeDiscount = item.price_min_before_discount || item.price_before_discount;
   106→    if (priceBeforeDiscount) {
   107→      const origVal = Array.isArray(priceBeforeDiscount) ? priceBeforeDiscount[0] : priceBeforeDiscount;
   108→      product.originalPrice = typeof origVal === "number" ? origVal : parseInt(String(origVal), 10);
   109→    }
   110→
   111→    // Historical price for original
   112→    if (!product.originalPrice && item.historical_price) {
   113→      const histVal = Array.isArray(item.historical_price) ? item.historical_price[0] : item.historical_price;
   114→      if (histVal && Number(histVal) > Number(product.price || 0)) {
   115→        product.originalPrice = typeof histVal === "number" ? histVal : parseInt(String(histVal), 10);
   116→      }
   117→    }
   118→
   119→    // Rating
   120→    if (item.item_rating) {
   121→      product.rating = item.item_rating.rating_star || 0;
   122→      const ratingCount = item.item_rating.rating_count || [];
   123→      product.reviewCount = Array.isArray(ratingCount)
   124→        ? ratingCount.reduce((sum: number, c: number) => sum + (c || 0), 0)
   125→        : 0;
   126→    }
   127→
   128→    // Sold count
   129→    if (item.historical_sold !== undefined && item.historical_sold !== null) {
   130→      product.soldCount = item.historical_sold;
   131→    } else if (item.sold !== undefined && item.sold !== null) {
   132→      product.soldCount = item.sold;
   133→    }
   134→
   135→    // Location / shop location
   136→    if (item.shop_location) {
   137→      product.location = item.shop_location;
   138→    } else if (item.shop_info) {
   139→      product.location = item.shop_info.shop_location || null;
   140→    }
   141→
   142→    // Category
   143→    if (item.catid) {
   144→      product.category = mapCategoryId(String(item.catid));
   145→    }
   146→
   147→    return product;
   148→  } catch (err) {
   149→    console.error("[scrape-shopee] Shopee API fetch failed:", err);
   150→    return null;
   151→  }
   152→}

   443→/**
   444→ * Fallback: parse HTML page for meta tags and JSON-LD
   445→ */
   446→async function scrapeHtmlPage(url: string): Promise<Record<string, unknown>> {
   447→  const product: Record<string, unknown> = {};
   448→
   449→  try {
   450→    const res = await fetch(url, {
   451→      headers: {
   452→        "User-Agent":
   453→          "Mozilla/5.0 (iPhone; CPU iPhone OS 17_0 like Mac OS X) AppleWebKit/605.1.15 (KHTML, like Gecko) Version/17.0 Mobile/15E148 Safari/604.1",
   454→        Accept: "text/html,application/xhtml+xml",
   455→        "Accept-Language": "id-ID,id;q=0.9,en-US;q=0.8,en;q=0.7",
   456→      },
   457→      signal: AbortSignal.timeout(20000),
   458→      redirect: "follow",
   459→    });
   460→    const pageContent = await res.text();
   461→
   462→    // JSON-LD
   463→    const jsonLdMatch = pageContent.match(
   464→      /<script\s+type=["']application\/ld\+json["'][^>]*>([\s\S]*?)<\/script>/gi
   465→    );
   466→    if (jsonLdMatch) {
   467→      for (const block of jsonLdMatch) {
   468→        try {
   469→          const inner = block.match(/<script[^>]*>([\s\S]*?)<\/script>/i);
   470→          if (inner) {
   471→            const parsed = JSON.parse(inner[1]);
   472→            if (parsed["@type"] === "Product" || parsed.name) {
   473→              product.title = parsed.name || "";
   474→              if (parsed.image) {
   475→                product.image = typeof parsed.image === "string" ? parsed.image : Array.isArray(parsed.image) ? parsed.image[0] : "";
   476→              }
   477→              if (parsed.offers) {
   478→                const offers = parsed.offers;
   479→                if (offers.price) product.price = parsePrice(offers.price);
   480→                if (offers.highPrice) product.originalPrice = parsePrice(offers.highPrice);
   481→              }
   482→              if (parsed.aggregateRating) {
   483→                product.rating = parseFloat(parsed.aggregateRating.ratingValue) || undefined;
   484→                product.reviewCount = parseInt(parsed.aggregateRating.reviewCount) || undefined;
   485→              }
   486→            }
   487→          }
   488→        } catch {}
   489→      }
   490→    }
   491→
   492→    // Meta tags
   493→    const ogTitle = pageContent.match(/<meta\s+property=["']og:title["']\s+content=["']([^"']+)["']/i);
   494→    if (ogTitle && !product.title) product.title = ogTitle[1];
   495→
   496→    const ogImage = pageContent.match(/<meta\s+property=["']og:image["']\s+content=["']([^"']+)["']/i);
   497→    if (ogImage && !product.image) product.image = ogImage[1];
   498→
   499→    const metaPrice = pageContent.match(/<meta\s+(?:property|name)=["']product:price:amount["']\s+content=["']([^"']+)["']/i);
   500→    if (metaPrice && !product.price) product.price = parsePrice(metaPrice[1]);
   501→
   502→    // Clean up title
   503→    if (product.title) {
   504→      product.title = String(product.title)
   505→        .replace(/\s*[-|–]\s*Shopee\s*Indonesia\s*/gi, "")
   506→        .replace(/\s*[-|–]\s*Shopee\s*/gi, "")
   507→        .trim();
   508→    }
   509→  } catch (err) {
   510→    console.error("[scrape-shopee] HTML scrape failed:", err);
   511→  }
   512→
   513→  return product;
   514→}

   516→/**
   517→ * Groq AI fallback extraction
   518→ */
   519→async function extractWithAI(pageContent: string, product: Record<string, unknown>): Promise<Record<string, unknown>> {
   520→  // Try ZAI LLM first (more reliable), then fall back to Groq
   521→  const existingTitle = product.title || "";
   522→  const existingPrice = product.price || "";
   523→  const existingImage = product.image || "";
   524→
   525→  // ZAI LLM approach
   526→  try {
   527→    const zai = await ZAI.create();
   528→    const prompt = `Kamu adalah extractor data produk Shopee Indonesia. Dari konten halaman Shopee berikut, extract info produk dan kembalikan HANYA JSON object dengan field ini:
   529→- title: nama produk (string)
   530→- image: URL foto produk utama (string)
   531→- price: harga dalam Rupiah, angka saja tanpa titik/koma (number)
   532→- originalPrice: harga asli sebelum diskon, angka saja, atau null (number|null)
   533→- rating: rating 0-5 (number)
   534→- reviewCount: jumlah review (number)
   535→- soldCount: jumlah terjual (number)
   536→- location: kota seller (string atau null)
   537→- category: salah satu dari Fashion, Beauty, Elektronik, Home, Gaming, Olahraga, Mainan, Otomotif (string)
   538→
   539→Data yang sudah diketahui: ${existingTitle ? `title="${existingTitle}"` : ""} ${existingPrice ? `price=${existingPrice}` : ""} ${existingImage ? `image="${existingImage}"` : ""}

   646→export async function POST(req: NextRequest) {
   647→  const authErr = await checkAuth(req);
   648→  if (authErr) return authErr;
   649→
   650→  try {
   651→    const body = await req.json();
   652→    let { url } = body;
   653→
   654→    if (!url) {
   655→      return NextResponse.json(
   656→        { error: "URL Shopee tidak valid" },
   657→        { status: 400 }
   658→      );
   659→    }
   660→
   661→    // SSRF Protection: Validate URL dengan proper parsing + domain allowlist
   662→    try {
   663→      const parsed = new URL(url);
   664→      const allowedHosts = [
   665→        "shopee.co.id",
   666→        "shope.ee",
   667→        "shopee.ph",
   668→        "shopee.sg",
   669→        "shopee.com.my",
   670→        "shopee.th",
   671→        "shopee.vn",
   672→        "shopee.com.br",
   673→        "shopee.com.mx",
   674→        "shopee.cl",
   675→        "shopee.pl",
   676→        "shopee.com.co",
   677→      ];
   678→      const isAllowed = allowedHosts.some(
   679→        (host) => parsed.hostname === host || parsed.hostname.endsWith("." + host)
   680→      );
   681→      if (!isAllowed) {
   682→        return NextResponse.json(
   683→          { error: "URL Shopee tidak valid — domain tidak diizinkan" },
   684→          { status: 400 }
   685→        );
   686→      }
   687→      // Only allow https: and http: schemes
   688→      if (parsed.protocol !== "https:" && parsed.protocol !== "http:") {
   689→        return NextResponse.json(
   690→          { error: "URL Shopee tidak valid — scheme tidak diizinkan" },
   691→          { status: 400 }
   692→        );
   693→      }
   694→    } catch {
   695→      return NextResponse.json(
   696→        { error: "URL Shopee tidak valid — format URL salah" },
   697→        { status: 400 }
   698→      );
   699→    }
   700→
   701→    // Resolve short URLs (shope.ee)
   702→    if (url.includes("shope.ee") || url.includes("shope.ee/")) {
   703→      const resolved = await resolveShortUrl(url);
   704→      if (resolved) url = resolved;
   705→    }
   706→
   707→    let product: Record<string, unknown> = {};
   708→
   709→    // Strategy 1: Use Shopee API v4 (best approach, but often blocked now)
   710→    const ids = extractIds(url);
   711→    if (ids) {
   712→      console.log("[scrape-shopee] Trying API v4, shopId:", ids.shopId, "itemId:", ids.itemId);
   713→      const apiResult = await fetchShopeeApi(ids.shopId, ids.itemId);
   714→      if (apiResult) {
   715→        product = apiResult;
   716→      }
   717→    }
   718→
   719→    // Strategy 2: HTML scrape if API didn't return enough data
   720→    if (!product.title || !product.image || !product.price) {
   721→      console.log("[scrape-shopee] API incomplete, trying HTML scrape...");
   722→      const htmlResult = await scrapeHtmlPage(url);
   723→      if (!product.title && htmlResult.title) product.title = htmlResult.title;
   724→      if (!product.image && htmlResult.image) product.image = htmlResult.image;
   725→      if (!product.price && htmlResult.price) product.price = htmlResult.price;
   726→      if (!product.originalPrice && htmlResult.originalPrice) product.originalPrice = htmlResult.originalPrice;
   727→      if (!product.rating && htmlResult.rating) product.rating = htmlResult.rating;
   728→      if (!product.reviewCount && htmlResult.reviewCount) product.reviewCount = htmlResult.reviewCount;
   729→      if (!product.soldCount && htmlResult.soldCount) product.soldCount = htmlResult.soldCount;
   730→      if (!product.location && htmlResult.location) product.location = htmlResult.location;
   731→      if (!product.category && htmlResult.category) product.category = htmlResult.category;
   732→    }
   733→
   734→    // Strategy 3: ZAI page_reader + AI extraction (most reliable for Shopee)
   735→    if (!product.title || !product.price) {
   736→      console.log("[scrape-shopee] Trying ZAI page_reader + AI extraction...");
   737→      try {
   738→        const zai = await ZAI.create();
   739→        const pageResult = await zai.functions.invoke("page_reader", { url });
   740→        const pageHtml = pageResult?.data?.html || "";
   741→        const pageTitle = pageResult?.data?.title || "";
   742→
   743→        if (pageHtml.length > 100) {
   744→          product = await extractWithAI(pageHtml, product);
   745→        }
   746→
   747→        // Jika masih belum dapat judul, coba dari page title
   748→        if (!product.title && pageTitle && pageTitle !== "Shopee Indonesia | Situs Belanja Online Terlengkap & Terpercaya") {
   749→          product.title = pageTitle.replace(/\s*[-|–]\s*Shopee\s*Indonesia\s*/gi, "").replace(/\s*[-|–]\s*Shopee\s*/gi, "").trim();
   750→        }
   751→      } catch (zaiErr) {
   752→        console.error("[scrape-shopee] ZAI page_reader failed:", zaiErr);
   753→      }
   754→    }
   755→
   756→    // Strategy 4: Groq AI fallback on HTML content (original approach)
   757→    if (!product.title || !product.price) {
   758→      console.log("[scrape-shopee] Trying Groq AI fallback...");
   759→      try {
   760→        const pageRes = await fetch(url, {
   761→          headers: {
   762→            "User-Agent":
   763→              "Mozilla/5.0 (iPhone; CPU iPhone OS 17_0 like Mac OS X) AppleWebKit/605.1.15 (KHTML, like Gecko) Version/17.0 Mobile/15E148 Safari/604.1",
   764→            Accept: "text/html,application/xhtml+xml",
   765→            "Accept-Language": "id-ID,id;q=0.9",
   766→          },
   767→          signal: AbortSignal.timeout(15000),
   768→          redirect: "follow",
   769→        });
   770→        const pageContent = await pageRes.text();
   771→        if (pageContent.length > 100) {
   772→          product = await extractWithAI(pageContent, product);
   773→        }
   774→      } catch {}
   775→    }
   776→
   777→    // Set defaults
   778→    if (!product.category) product.category = "Fashion";
   779→    if (!product.rating) product.rating = 4.5;
   780→    if (!product.reviewCount) product.reviewCount = 0;
   781→    if (!product.soldCount) product.soldCount = 0;
   782→
   783→    const price = Number(product.price) || 0;
   784→    const origPrice = Number(product.originalPrice) || 0;
   785→    const discountPercent = origPrice > price ? Math.round(((origPrice - price) / origPrice) * 100) : null;
   786→
   787→    // Final validation - at minimum we need a title
   788→    if (!product.title) {
   789→      return NextResponse.json(
   790→        { error: "Gagal mengambil data produk. Coba paste link produk yang lain." },
   791→        { status: 422 }
   792→      );
   793→    }
   794→
   795→    return NextResponse.json({
   796→      success: true,
   797→      product: {
   798→        title: String(product.title),
   799→        image: String(product.image || ""),
   800→        price,
   801→        originalPrice: origPrice > 0 ? origPrice : null,
   802→        discountPercent,
   803→        rating: Number(product.rating) || 4.5,
   804→        reviewCount: Number(product.reviewCount) || 0,
   805→        soldCount: Number(product.soldCount) || 0,
   806→        location: product.location ? String(product.location) : null,
   807→        category: String(product.category || "Fashion"),
   808→      },
   809→    });
   810→  } catch (err) {
   811→    console.error("[scrape-shopee] Error:", err);
   812→    return NextResponse.json(
   813→      { error: "Gagal scrape produk Shopee. Coba lagi nanti." },
   814→      { status: 500 }
   815→    );
   816→  }
   817→}

   819→function parsePrice(val: unknown): number | null {
   820→  if (val === null || val === undefined) return null;
   821→  const str = String(val).replace(/[^\d]/g, "");
   822→  const num = parseInt(str, 10);
   823→  return isNaN(num) ? null : num;
   824→}
