    17→/**
    18→ * Extract shopid and itemid from various Shopee URL formats:
    19→ * - https://shopee.co.id/product-name-i.SHOPID.ITEMID
    20→ * - https://shopee.co.id/product-name-i.SHOPID.ITEMID?...
    21→ * - https://shope.ee/xxx (short URL, will be resolved)
    22→ */
    23→function extractIds(url: string): { shopId: number; itemId: number } | null {
    24→  // Pattern: -i.SHOPID.ITEMID at end of path
    25→  const match = url.match(/-i\.(\d+)\.(\d+)/);
    26→  if (match) {
    27→    return { shopId: parseInt(match[1], 10), itemId: parseInt(match[2], 10) };
    28→  }
    29→  return null;
    30→}

    51→/**
    52→ * Fetch product data using Shopee's internal API v4
    53→ */
    54→async function fetchShopeeApi(shopId: number, itemId: number): Promise<Record<string, unknown> | null> {
    55→  const apiUrl = `https://shopee.co.id/api/v4/item/get?itemid=${itemId}&shopid=${shopId}`;
    56→
    57→  try {
    58→    const res = await fetch(apiUrl, {
    59→      headers: {
    60→        "User-Agent":
    61→          "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/125.0.0.0 Safari/537.36",
    62→        Accept: "application/json",
    63→        "Accept-Language": "id-ID,id;q=0.9,en-US;q=0.8,en;q=0.7",
    64→        "X-Shopee-Language": "id",
    65→        "X-API-SOURCE": "pc",
    66→        Referer: `https://shopee.co.id/`,
    67→        "af-ac-enc-dat": "null",
    68→      },
    69→      signal: AbortSignal.timeout(15000),
    70→    });
    71→
    72→    if (!res.ok) {
    73→      console.error("[scrape-shopee] Shopee API status:", res.status);
    74→      return null;
    75→    }
    76→
    77→    const data = await res.json();
    78→
    79→    if (data.error) {
    80→      console.error("[scrape-shopee] Shopee API error:", data.error, data.msg || "");
    81→      return null;
    82→    }
    83→
    84→    const item = data.data;
    85→    if (!item) return null;
    86→
    87→    const product: Record<string, unknown> = {};
    88→
    89→    // Title
    90→    product.title = item.name || "";
    91→
    92→    // Image - Shopee returns image field as "xxx" which needs prefix
    93→    if (item.image) {
    94→      product.image = `https://down-id.img.susercontent.com/file/${item.image}`;
    95→    } else if (item.images && Array.isArray(item.images) && item.images.length > 0) {
    96→      product.image = `https://down-id.img.susercontent.com/file/${item.images[0]}`;
    97→    }
    98→
    99→    // Price - Shopee API returns prices in minor units (need to handle)
   100→    // price_min and price_max are in the smallest currency unit
   101→    // For IDR, Shopee sometimes returns the actual price (no conversion needed for IDR)
   102→    const priceMin = item.price_min || item.price;
   103→    const priceMax = item.price_max || item.price;
   104→
   105→    if (priceMin) {
   106→      // Shopee API returns price as array [min, max] or single number
   107→      const priceVal = Array.isArray(priceMin) ? priceMin[0] : priceMin;
   108→      // If price is unreasonably large (above 10M IDR per unit), it might be in minor units
   109→      product.price = typeof priceVal === "number" ? priceVal : parseInt(String(priceVal), 10);
   110→    }
   111→
   112→    // Original price (before discount)
   113→    const priceBeforeDiscount = item.price_min_before_discount || item.price_before_discount;
   114→    if (priceBeforeDiscount) {
   115→      const origVal = Array.isArray(priceBeforeDiscount) ? priceBeforeDiscount[0] : priceBeforeDiscount;
   116→      product.originalPrice = typeof origVal === "number" ? origVal : parseInt(String(origVal), 10);
   117→    }
   118→
   119→    // Historical price for original
   120→    if (!product.originalPrice && item.historical_price) {
   121→      const histVal = Array.isArray(item.historical_price) ? item.historical_price[0] : item.historical_price;
   122→      if (histVal && Number(histVal) > Number(product.price || 0)) {
   123→        product.originalPrice = typeof histVal === "number" ? histVal : parseInt(String(histVal), 10);
   124→      }
   125→    }
   126→
   127→    // Rating
   128→    if (item.item_rating) {
   129→      product.rating = item.item_rating.rating_star || 0;
   130→      const ratingCount = item.item_rating.rating_count || [];
   131→      product.reviewCount = Array.isArray(ratingCount)
   132→        ? ratingCount.reduce((sum: number, c: number) => sum + (c || 0), 0)
   133→        : 0;
   134→    }
   135→
   136→    // Sold count
   137→    if (item.historical_sold !== undefined && item.historical_sold !== null) {
   138→      product.soldCount = item.historical_sold;
   139→    } else if (item.sold !== undefined && item.sold !== null) {
   140→      product.soldCount = item.sold;
   141→    }
   142→
   143→    // Location / shop location
   144→    if (item.shop_location) {
   145→      product.location = item.shop_location;
   146→    } else if (item.shop_info) {
   147→      product.location = item.shop_info.shop_location || null;
   148→    }
   149→
   150→    // Category
   151→    if (item.catid) {
   152→      product.category = mapCategoryId(String(item.catid));
   153→    }
   154→
   155→    return product;
   156→  } catch (err) {
   157→    console.error("[scrape-shopee] Shopee API fetch failed:", err);
   158→    return null;
   159→  }
   160→}

   451→/**
   452→ * Fallback: parse HTML page for meta tags and JSON-LD
   453→ */
   454→async function scrapeHtmlPage(url: string): Promise<Record<string, unknown>> {
   455→  const product: Record<string, unknown> = {};
   456→
   457→  try {
   458→    const res = await fetch(url, {
   459→      headers: {
   460→        "User-Agent":
   461→          "Mozilla/5.0 (iPhone; CPU iPhone OS 17_0 like Mac OS X) AppleWebKit/605.1.15 (KHTML, like Gecko) Version/17.0 Mobile/15E148 Safari/604.1",
   462→        Accept: "text/html,application/xhtml+xml",
   463→        "Accept-Language": "id-ID,id;q=0.9,en-US;q=0.8,en;q=0.7",
   464→      },
   465→      signal: AbortSignal.timeout(20000),
   466→      redirect: "follow",
   467→    });
   468→    const pageContent = await res.text();
   469→
   470→    // JSON-LD
   471→    const jsonLdMatch = pageContent.match(
   472→      /<script\s+type=["']application\/ld\+json["'][^>]*>([\s\S]*?)<\/script>/gi
   473→    );
   474→    if (jsonLdMatch) {
   475→      for (const block of jsonLdMatch) {
   476→        try {
   477→          const inner = block.match(/<script[^>]*>([\s\S]*?)<\/script>/i);
   478→          if (inner) {
   479→            const parsed = JSON.parse(inner[1]);
   480→            if (parsed["@type"] === "Product" || parsed.name) {
   481→              product.title = parsed.name || "";
   482→              if (parsed.image) {
   483→                product.image = typeof parsed.image === "string" ? parsed.image : Array.isArray(parsed.image) ? parsed.image[0] : "";
   484→              }
   485→              if (parsed.offers) {
   486→                const offers = parsed.offers;
   487→                if (offers.price) product.price = parsePrice(offers.price);
   488→                if (offers.highPrice) product.originalPrice = parsePrice(offers.highPrice);
   489→              }
   490→              if (parsed.aggregateRating) {
   491→                product.rating = parseFloat(parsed.aggregateRating.ratingValue) || undefined;
   492→                product.reviewCount = parseInt(parsed.aggregateRating.reviewCount) || undefined;
   493→              }
   494→            }
   495→          }
   496→        } catch {}
   497→      }
   498→    }
   499→
   500→    // Meta tags
   501→    const ogTitle = pageContent.match(/<meta\s+property=["']og:title["']\s+content=["']([^"']+)["']/i);
   502→    if (ogTitle && !product.title) product.title = ogTitle[1];
   503→
   504→    const ogImage = pageContent.match(/<meta\s+property=["']og:image["']\s+content=["']([^"']+)["']/i);
   505→    if (ogImage && !product.image) product.image = ogImage[1];
   506→
   507→    const metaPrice = pageContent.match(/<meta\s+(?:property|name)=["']product:price:amount["']\s+content=["']([^"']+)["']/i);
   508→    if (metaPrice && !product.price) product.price = parsePrice(metaPrice[1]);
   509→
   510→    // Clean up title
   511→    if (product.title) {
   512→      product.title = String(product.title)
   513→        .replace(/\s*[-|–]\s*Shopee\s*Indonesia\s*/gi, "")
   514→        .replace(/\s*[-|–]\s*Shopee\s*/gi, "")
   515→        .trim();
   516→    }
   517→  } catch (err) {
   518→    console.error("[scrape-shopee] HTML scrape failed:", err);
   519→  }
   520→
   521→  return product;
   522→}

   524→/**
   525→ * Groq AI fallback extraction
   526→ */
   527→async function extractWithAI(pageContent: string, product: Record<string, unknown>): Promise<Record<string, unknown>> {
   528→  // Try ZAI LLM first (more reliable), then fall back to Groq
   529→  const existingTitle = product.title || "";
   530→  const existingPrice = product.price || "";
   531→  const existingImage = product.image || "";
   532→
   533→  // ZAI LLM approach
   534→  try {
   535→    const zai = await ZAI.create();
   536→    const prompt = `Kamu adalah extractor data produk Shopee Indonesia. Dari konten halaman Shopee berikut, extract info produk dan kembalikan HANYA JSON object dengan field ini:
   537→- title: nama produk (string)
   538→- image: URL foto produk utama (string)
   539→- price: harga dalam Rupiah, angka saja tanpa titik/koma (number)
   540→- originalPrice: harga asli sebelum diskon, angka saja, atau null (number|null)
   541→- rating: rating 0-5 (number)
   542→- reviewCount: jumlah review (number)

   654→export async function POST(req: NextRequest) {
   655→  const authErr = checkAuth(req);
   656→  if (authErr) return authErr;
   657→
   658→  try {
   659→    const body = await req.json();
   660→    let { url } = body;
   661→
   662→    if (!url || (!url.includes("shopee") && !url.includes("shope"))) {
   663→      return NextResponse.json(
   664→        { error: "URL Shopee tidak valid" },
   665→        { status: 400 }
   666→      );
   667→    }
   668→
   669→    // Resolve short URLs (shope.ee)
   670→    if (url.includes("shope.ee") || url.includes("shope.ee/")) {
   671→      const resolved = await resolveShortUrl(url);
   672→      if (resolved) url = resolved;
   673→    }
   674→
   675→    let product: Record<string, unknown> = {};
   676→
   677→    // Strategy 1: Use Shopee API v4 (best approach, but often blocked now)
   678→    const ids = extractIds(url);
   679→    if (ids) {
   680→      console.log("[scrape-shopee] Trying API v4, shopId:", ids.shopId, "itemId:", ids.itemId);
   681→      const apiResult = await fetchShopeeApi(ids.shopId, ids.itemId);
   682→      if (apiResult) {
   683→        product = apiResult;
   684→      }
   685→    }
   686→
   687→    // Strategy 2: HTML scrape if API didn't return enough data
   688→    if (!product.title || !product.image || !product.price) {
   689→      console.log("[scrape-shopee] API incomplete, trying HTML scrape...");
   690→      const htmlResult = await scrapeHtmlPage(url);
   691→      if (!product.title && htmlResult.title) product.title = htmlResult.title;
   692→      if (!product.image && htmlResult.image) product.image = htmlResult.image;
   693→      if (!product.price && htmlResult.price) product.price = htmlResult.price;
   694→      if (!product.originalPrice && htmlResult.originalPrice) product.originalPrice = htmlResult.originalPrice;
   695→      if (!product.rating && htmlResult.rating) product.rating = htmlResult.rating;
   696→      if (!product.reviewCount && htmlResult.reviewCount) product.reviewCount = htmlResult.reviewCount;
   697→      if (!product.soldCount && htmlResult.soldCount) product.soldCount = htmlResult.soldCount;
   698→      if (!product.location && htmlResult.location) product.location = htmlResult.location;
   699→      if (!product.category && htmlResult.category) product.category = htmlResult.category;
   700→    }
   701→
   702→    // Strategy 3: ZAI page_reader + AI extraction (most reliable for Shopee)
   703→    if (!product.title || !product.price) {
   704→      console.log("[scrape-shopee] Trying ZAI page_reader + AI extraction...");
   705→      try {
   706→        const zai = await ZAI.create();
   707→        const pageResult = await zai.functions.invoke("page_reader", { url });
   708→        const pageHtml = pageResult?.data?.html || "";
   709→        const pageTitle = pageResult?.data?.title || "";
   710→
   711→        if (pageHtml.length > 100) {
   712→          product = await extractWithAI(pageHtml, product);
   713→        }
   714→
   715→        // Jika masih belum dapat judul, coba dari page title
   716→        if (!product.title && pageTitle && pageTitle !== "Shopee Indonesia | Situs Belanja Online Terlengkap & Terpercaya") {
   717→          product.title = pageTitle.replace(/\s*[-|–]\s*Shopee\s*Indonesia\s*/gi, "").replace(/\s*[-|–]\s*Shopee\s*/gi, "").trim();
   718→        }
   719→      } catch (zaiErr) {
   720→        console.error("[scrape-shopee] ZAI page_reader failed:", zaiErr);
   721→      }
   722→    }
   723→
   724→    // Strategy 4: Groq AI fallback on HTML content (original approach)
   725→    if (!product.title || !product.price) {
   726→      console.log("[scrape-shopee] Trying Groq AI fallback...");
   727→      try {
   728→        const pageRes = await fetch(url, {
   729→          headers: {
   730→            "User-Agent":
   731→              "Mozilla/5.0 (iPhone; CPU iPhone OS 17_0 like Mac OS X) AppleWebKit/605.1.15 (KHTML, like Gecko) Version/17.0 Mobile/15E148 Safari/604.1",
   732→            Accept: "text/html,application/xhtml+xml",
   733→            "Accept-Language": "id-ID,id;q=0.9",
   734→          },
   735→          signal: AbortSignal.timeout(15000),
   736→          redirect: "follow",
   737→        });
   738→        const pageContent = await pageRes.text();
   739→        if (pageContent.length > 100) {
   740→          product = await extractWithAI(pageContent, product);
   741→        }
   742→      } catch {}
   743→    }
   744→
   745→    // Set defaults
   746→    if (!product.category) product.category = "Fashion";
   747→    if (!product.rating) product.rating = 4.5;
   748→    if (!product.reviewCount) product.reviewCount = 0;
   749→    if (!product.soldCount) product.soldCount = 0;
   750→
   751→    const price = Number(product.price) || 0;
   752→    const origPrice = Number(product.originalPrice) || 0;
   753→    const discountPercent = origPrice > price ? Math.round(((origPrice - price) / origPrice) * 100) : null;
   754→
   755→    // Final validation - at minimum we need a title
   756→    if (!product.title) {
   757→      return NextResponse.json(
   758→        { error: "Gagal mengambil data produk. Coba paste link produk yang lain." },
   759→        { status: 422 }
   760→      );
   761→    }
   762→
   763→    return NextResponse.json({
   764→      success: true,
   765→      product: {
   766→        title: String(product.title),
   767→        image: String(product.image || ""),
   768→        price,
   769→        originalPrice: origPrice > 0 ? origPrice : null,
   770→        discountPercent,
   771→        rating: Number(product.rating) || 4.5,
   772→        reviewCount: Number(product.reviewCount) || 0,
   773→        soldCount: Number(product.soldCount) || 0,
   774→        location: product.location ? String(product.location) : null,
   775→        category: String(product.category || "Fashion"),
   776→      },
   777→    });
   778→  } catch (err) {
   779→    console.error("[scrape-shopee] Error:", err);
   780→    return NextResponse.json(
   781→      { error: "Gagal scrape produk Shopee. Coba lagi nanti." },
   782→      { status: 500 }
   783→    );
   784→  }
   785→}

   787→function parsePrice(val: unknown): number | null {
   788→  if (val === null || val === undefined) return null;
   789→  const str = String(val).replace(/[^\d]/g, "");
   790→  const num = parseInt(str, 10);
   791→  return isNaN(num) ? null : num;
   792→}
