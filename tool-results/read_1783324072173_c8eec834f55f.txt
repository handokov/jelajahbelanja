   130тЖТ   130тЖТasync function saveAllImagesToLocal(products) {
   131тЖТ   131тЖТ  const needSave = products.filter(p => p.image && !p.image.includes('localhost:3000'));
   132тЖТ   132тЖТ  if (needSave.length === 0) return { success: 0, failed: 0, total: 0 };
   133тЖТ   133тЖТ  
   134тЖТ   134тЖТ  // Coba batch endpoint dulu (lebih cepat)
   135тЖТ   135тЖТ  try {
   136тЖТ   136тЖТ    const urls = needSave.map(p => p.image);
   137тЖТ   137тЖТ    const resp = await fetch(`${IMAGE_SERVER}/batch`, {
   138тЖТ   138тЖТ      method: 'POST',
   139тЖТ   139тЖТ      headers: { 'Content-Type': 'application/json' },
   140тЖТ   140тЖТ      body: JSON.stringify({ urls }),
   141тЖТ   141тЖТ      signal: AbortSignal.timeout(60000),
   142тЖТ   142тЖТ    });
   143тЖТ   143тЖТ    const data = await resp.json();
   144тЖТ   144тЖТ    if (data.results) {
   145тЖТ   145тЖТ      let successCount = 0;
   146тЖТ   146тЖТ      for (let i = 0; i < needSave.length; i++) {
   147тЖТ   147тЖТ        if (data.results[i]?.success) {
   148тЖТ   148тЖТ          needSave[i].image = data.results[i].localUrl;
   149тЖТ   149тЖТ          successCount++;
   150тЖТ   150тЖТ        }
   151тЖТ   151тЖТ      }
   152тЖТ   152тЖТ      return { success: successCount, failed: needSave.length - successCount, total: needSave.length };
   153тЖТ   153тЖТ    }
   154тЖТ   154тЖТ  } catch {}
   155тЖТ   155тЖТ  
   156тЖТ   156тЖТ  // Fallback: download satu-satu
   157тЖТ   157тЖТ  let successCount = 0;
   158тЖТ   158тЖТ  for (const p of needSave) {
   159тЖТ   159тЖТ    const newUrl = await saveImageToLocal(p.image);
   160тЖТ   160тЖТ    if (newUrl !== p.image) {
   161тЖТ   161тЖТ      p.image = newUrl;
   162тЖТ   162тЖТ      successCount++;
   163тЖТ   163тЖТ    }
   164тЖТ   164тЖТ  }
   165тЖТ   165тЖТ  return { success: successCount, failed: needSave.length - successCount, total: needSave.length };
   166тЖТ   166тЖТ}

   259тЖТ   259тЖТfunction parseCSVLine(line) {
   260тЖТ   260тЖТ  const fields = [];
   261тЖТ   261тЖТ  let current = '';
   262тЖТ   262тЖТ  let inQuotes = false;
   263тЖТ   263тЖТ
   264тЖТ   264тЖТ  for (let i = 0; i < line.length; i++) {
   265тЖТ   265тЖТ    const ch = line[i];
   266тЖТ   266тЖТ    if (inQuotes) {
   267тЖТ   267тЖТ      if (ch === '"') {
   268тЖТ   268тЖТ        if (i + 1 < line.length && line[i + 1] === '"') {
   269тЖТ   269тЖТ          current += '"';
   270тЖТ   270тЖТ          i++;
   271тЖТ   271тЖТ        } else {
   272тЖТ   272тЖТ          inQuotes = false;
   273тЖТ   273тЖТ        }
   274тЖТ   274тЖТ      } else {
   275тЖТ   275тЖТ        current += ch;
   276тЖТ   276тЖТ      }
   277тЖТ   277тЖТ    } else {
   278тЖТ   278тЖТ      if (ch === '"') {
   279тЖТ   279тЖТ        inQuotes = true;
   280тЖТ   280тЖТ      } else if (ch === ',') {
   281тЖТ   281тЖТ        fields.push(current);
   282тЖТ   282тЖТ        current = '';
   283тЖТ   283тЖТ      } else {
   284тЖТ   284тЖТ        current += ch;
   285тЖТ   285тЖТ      }
   286тЖТ   286тЖТ    }
   287тЖТ   287тЖТ  }
   288тЖТ   288тЖТ  fields.push(current);
   289тЖТ   289тЖТ  return fields;
   290тЖТ   290тЖТ}

   395тЖТ   395тЖТ    let resolved = false;

   427тЖТ   427тЖТ            resolved = true;
